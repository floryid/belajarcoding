3. Menyimpan Data dengan LocalStorage

Gunakan LocalStorage untuk menyimpan data di browser:


Contoh kode JavaScript:

function saveData() {
    const data = document.getElementById('dataStorage').value;
    if (data) {
        localStorage.setItem('userData', data);
        alert('Data disimpan!');
    } else {
        alert('Masukkan data terlebih dahulu.');
    }
}

window.onload = function() {
    const savedData = localStorage.getItem('userData');
    if (savedData) {
        document.getElementById('dataStorage').value = savedData;
    }
}
